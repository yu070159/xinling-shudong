-- 情绪年轮：每日心情记录表
CREATE TABLE IF NOT EXISTS mood_logs (
  id BIGSERIAL PRIMARY KEY,
  user_id UUID REFERENCES auth.users NOT NULL,
  mood_type TEXT NOT NULL CHECK (mood_type IN ('开心', '平静', '难过', '焦虑', '生气')),
  note TEXT,
  created_at DATE NOT NULL DEFAULT CURRENT_DATE,
  UNIQUE (user_id, created_at)
);

-- RLS: 用户只能读写自己的记录
ALTER TABLE mood_logs ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Users can read own mood_logs" ON mood_logs
  FOR SELECT USING (auth.uid() = user_id);

CREATE POLICY "Users can insert own mood_logs" ON mood_logs
  FOR INSERT WITH CHECK (auth.uid() = user_id);

CREATE POLICY "Users can update own mood_logs" ON mood_logs
  FOR UPDATE USING (auth.uid() = user_id);

CREATE POLICY "Users can delete own mood_logs" ON mood_logs
  FOR DELETE USING (auth.uid() = user_id);
```

async function loadTodayRecord() {
  if (!currentUser) return;
  const today = new Date().toISOString().slice(0, 10);
  const { data } = await sb.from('mood_logs')
    .select('mood_type, note')
    .eq('user_id', currentUser.id)
    .eq('created_at', today)
    .maybeSingle();
  todayRecord = data;
  return data;
}

function renderCheckin() {
  const section = document.getElementById('checkinSection');
  if (!section) return;

  if (todayRecord) {
    const cfg = MOOD_CONFIG[todayRecord.mood_type];
    section.innerHTML = `
      <div class="mood-checkin-done">
        今日心情
        <div class="today-mood">
          <span style="font-size:32px">${cfg.emoji}</span>
          <span style="color:${cfg.color};font-weight:600;margin-left:8px">${todayRecord.mood_type}</span>
        </div>
        ${todayRecord.note ? `<div style="font-size:13px;color:var(--text-muted);margin-top:4px">"${escapeHtml(todayRecord.note)}"</div>` : ''}
        <button class="mood-modify-btn" id="modifyMoodBtn">修改心情</button>
      </div>`;
    document.getElementById('modifyMoodBtn').addEventListener('click', () => {
      todayRecord = null;
      renderCheckin();
    });
  } else {
    let selectedMood = null;
    section.innerHTML = `
      <div class="mood-checkin-title">今天你的心情是？</div>
      <div class="mood-options" id="moodOptions"></div>
      <textarea class="mood-note" id="moodNote" placeholder="想说点什么...（选填）" maxlength="100"></textarea>
      <br>
      <button class="mood-submit-btn" id="moodSubmitBtn" disabled>记录今日心情</button>
      <div class="mood-msg" id="moodMsg"></div>`;

    const optionsContainer = document.getElementById('moodOptions');
    MOOD_TYPES.forEach(type => {
      const cfg = MOOD_CONFIG[type];
      const btn = document.createElement('div');
      btn.className = 'mood-option';
      btn.innerHTML = `<span class="mood-emoji">${cfg.emoji}</span><span class="mood-label">${type}</span>`;
      btn.addEventListener('click', () => {
        optionsContainer.querySelectorAll('.mood-option').forEach(o => o.classList.remove('selected'));
        btn.classList.add('selected');
        selectedMood = type;
        document.getElementById('moodSubmitBtn').disabled = false;
      });
      optionsContainer.appendChild(btn);
    });

    document.getElementById('moodSubmitBtn').addEventListener('click', async () => {
      if (!selectedMood) return;
      const btn = document.getElementById('moodSubmitBtn');
      const msg = document.getElementById('moodMsg');
      const note = document.getElementById('moodNote').value.trim();
      btn.disabled = true;
      btn.textContent = '正在记录...';
      msg.className = 'mood-msg';
      msg.textContent = '';

      const today = new Date().toISOString().slice(0, 10);
      const { data: existing } = await sb.from('mood_logs')
        .select('id').eq('user_id', currentUser.id).eq('created_at', today).maybeSingle();

      let error;
      if (existing) {
        const { error: e } = await sb.from('mood_logs')
          .update({ mood_type: selectedMood, note })
          .eq('id', existing.id);
        error = e;
      } else {
        const { error: e } = await sb.from('mood_logs')
          .insert({ user_id: currentUser.id, mood_type: selectedMood, note, created_at: today });
        error = e;
      }

      if (error) {
        msg.className = 'mood-msg error';
        msg.textContent = '记录失败，请稍后重试';
        btn.disabled = false;
        btn.textContent = '记录今日心情';
      } else {
        msg.className = 'mood-msg success';
        msg.textContent = '叶子已经放进年轮啦 🌿';
        todayRecord = { mood_type: selectedMood, note };
        setTimeout(() => { renderCheckin(); renderRing(); renderLegend(); }, 600);
      }
    });
  }
}

function createGinkgoLeaf(size, color, cx, cy, angle, opacity = 1) {
  const g = document.createElementNS('http://www.w3.org/2000/svg', 'g');
  g.setAttribute('transform', `translate(${cx},${cy}) rotate(${angle}) scale(${size / 36})`);
  g.style.transition = 'transform 0.2s ease';
  g.style.cursor = 'pointer';

  const path = document.createElementNS('http://www.w3.org/2000/svg', 'path');
  path.setAttribute('d', GINKGO_PATH);
  path.setAttribute('fill', color);
  path.setAttribute('opacity', opacity);
  g.appendChild(path);

  return g;
}

async function renderRing() {
  const wrap = document.getElementById('ringSvgWrap');
  const label = document.getElementById('ringYearLabel');
  if (!wrap || !label) return;
  label.textContent = currentYear + '年';

  if (!currentUser) {
    wrap.innerHTML = '<div class="mood-empty">请先登录以查看情绪年轮</div>';
    return;
  }

  // 加载当前年份 ± 1 年数据（确保12个月完整覆盖）
  const startDate = `${currentYear}-01-01`;
  const endDate = `${currentYear}-12-31`;
  const { data, error } = await sb.from('mood_logs')
    .select('mood_type, note, created_at')
    .eq('user_id', currentUser.id)
    .gte('created_at', startDate)
    .lte('created_at', endDate)
    .order('created_at', { ascending: true });

  if (error) {
    wrap.innerHTML = '<div class="mood-empty">加载失败，请刷新重试</div>';
    return;
  }

  const moodMap = {};
  if (data) data.forEach(r => { moodMap[r.created_at] = { mood_type: r.mood_type, note: r.note }; });

  const yearRecords = [];
  for (let m = 0; m < 12; m++) {
    const daysInMonth = new Date(currentYear, m + 1, 0).getDate();
    const monthData = [];
    for (let d = 1; d <= daysInMonth; d++) {
      const dateStr = `${currentYear}-${String(m + 1).padStart(2, '0')}-${String(d).padStart(2, '0')}`;
      monthData.push(moodMap[dateStr] || null);
    }
    yearRecords.push(monthData);
  }

  const hasAnyRecord = data && data.length > 0;
  if (!hasAnyRecord) {
    wrap.innerHTML = `<div class="mood-empty">${currentYear}年还没有记录心情<br>从今天开始，种下你的第一片叶子吧 🌱</div>`;
    return;
  }

  // SVG 参数
  const svgSize = 500;
  const cx = svgSize / 2;
  const cy = svgSize / 2;
  const outerRadius = 220;
  const ringGap = 3;
  const ringWidth = (outerRadius - 30) / 12 - ringGap;

  let svg = `<svg width="${svgSize}" height="${svgSize}" viewBox="0 0 ${svgSize} ${svgSize}">`;
  svg += `<circle cx="${cx}" cy="${cy}" r="22" fill="var(--bg-card)" stroke="var(--border-light)" stroke-width="1"/>`;
  svg += `<text x="${cx}" y="${cy + 4}" text-anchor="middle" font-size="11" fill="var(--accent)">${currentYear}</text>`;

  const tooltip = document.getElementById('leafTooltip');

  yearRecords.forEach((monthData, monthIndex) => {
    const ringR = 30 + (monthIndex + 1) * (ringWidth + ringGap);
    const daysCount = monthData.length;

    monthData.forEach((record, dayIndex) => {
      const angle = (dayIndex / daysCount) * 2 * Math.PI - Math.PI / 2;
      const lx = cx + ringR * Math.cos(angle);
      const ly = cy + ringR * Math.sin(angle);
      const deg = (angle * 180) / Math.PI + 90;

      const color = record
        ? MOOD_CONFIG[record.mood_type].color
        : 'rgba(180, 170, 155, 0.2)';
      const opacity = record ? 0.9 : 0.3;
      const dateStr = `${currentYear}-${String(monthIndex + 1).padStart(2, '0')}-${String(dayIndex + 1).padStart(2, '0')}`;

      svg += `<g transform="translate(${lx},${ly}) rotate(${deg}) scale(0.5)" style="cursor:pointer;transition:transform 0.2s ease"`;
      if (record) {
        const tooltipText = `${formatDate(dateStr)} · ${record.mood_type}${record.note ? '\\n' + escapeHtml(record.note) : ''}`;
        svg += ` onmouseenter="var t=document.getElementById('leafTooltip');t.innerHTML=\\'${tooltipText.replace(/'/g, "\\'")}\\';t.classList.add('show')"`;
        svg += ` onmousemove="var t=document.getElementById('leafTooltip');t.style.left=(event.clientX+12)+\\'px\\';t.style.top=(event.clientY-10)+\\'px\\'"`;
        svg += ` onmouseleave="document.getElementById('leafTooltip').classList.remove('show')"`;
      }
      svg += '>';
      svg += `<path d="${GINKGO_PATH}" fill="${color}" opacity="${opacity}"/>`;
      svg += '</g>';
    });

    // 月份标签（在每圈起始位置）
    const labelAngle = -Math.PI / 2;
    const labelR = ringR + ringWidth / 2;
    const labelX = cx + labelR * Math.cos(labelAngle) - 16;
    const labelY = cy + labelR * Math.sin(labelAngle) + 4;
    svg += `<text x="${labelX}" y="${labelY}" font-size="10" fill="var(--text-muted)">${monthIndex + 1}月</text>`;
  });

  svg += '</svg>';
  wrap.innerHTML = svg;
}

function renderLegend() {

async function loadAndRenderStats() {
  if (!currentUser) return;
  const yearStart = `${currentYear}-01-01`;
  const yearEnd = `${currentYear}-12-31`;
  const { data } = await sb.from('mood_logs')
    .select('mood_type')
    .eq('user_id', currentUser.id)
    .gte('created_at', yearStart)
    .lte('created_at', yearEnd);

  const stats = document.getElementById('moodStats');
  if (!stats) return;
  if (!data || data.length === 0) {
    stats.textContent = '';
    return;
  }
  const counts = {};
  data.forEach(r => { counts[r.mood_type] = (counts[r.mood_type] || 0) + 1; });
  const total = data.length;
  const parts = MOOD_TYPES.filter(t => counts[t]).map(t => `${MOOD_CONFIG[t].emoji}${t} ${counts[t]}天`);
  stats.textContent = `${currentYear}年已记录 ${total} 天 · ` + parts.join(' · ');
}

  // 年份切换
  document.getElementById('ringPrev').addEventListener('click', async () => {
    currentYear--;
    await renderRing();
    await loadAndRenderStats();
  });
  document.getElementById('ringNext').addEventListener('click', async () => {
    currentYear++;
    await renderRing();
    await loadAndRenderStats();
  });
});
```
